■ログインユーザーのみに操作を許可

drop policy if exists "booktype_master insert : only own user_id" on booktype_master;
drop policy if exists "booktype_master delete : only own user_id" on booktype_master;
drop policy if exists "booktype_master update : only own user_id" on booktype_master;
drop policy if exists "booktype_master select : only own user_id" on booktype_master;

create policy "booktype_master insert : only own user_id"
on "public"."booktype_master"
as PERMISSIVE
for INSERT
to public
with check (
  (auth.uid() = user_id)
);
create policy "booktype_master delete : only own user_id"
on "public"."booktype_master"
as PERMISSIVE
for DELETE
to public
using (
  (auth.uid() = user_id)
);
create policy "booktype_master update : only own user_id"
on "public"."booktype_master"
as PERMISSIVE
for UPDATE
to public
using (
  (auth.uid() = user_id)
);
create policy "booktype_master select : only own user_id"
on "public"."booktype_master"
as PERMISSIVE
for SELECT
to public
using (
  (auth.uid() = user_id)
);
