drop policy if exists "bookclass_master insert : only own user_id" on bookclass_master;
drop policy if exists "bookclass_master delete : only own user_id" on bookclass_master;
drop policy if exists "bookclass_master update : only own user_id" on bookclass_master;
drop policy if exists "bookclass_master select : only own user_id" on bookclass_master;

create policy "bookclass_master insert : only own user_id"
on "public"."bookclass_master"
as PERMISSIVE
for INSERT
to public
with check (
  (auth.uid() = user_id)
);
create policy "bookclass_master delete : only own user_id"
on "public"."bookclass_master"
as PERMISSIVE
for DELETE
to public
using (
  (auth.uid() = user_id)
);
create policy "bookclass_master update : only own user_id"
on "public"."bookclass_master"
as PERMISSIVE
for UPDATE
to public
using (
  (auth.uid() = user_id)
);
create policy "bookclass_master select : only own user_id"
on "public"."bookclass_master"

to public
using (
  (auth.uid() = user_id)
