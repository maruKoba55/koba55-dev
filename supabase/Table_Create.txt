DROP TABLE IF EXISTS profile;
DROP TABLE IF EXISTS book_note;
DROP TABLE IF EXISTS book_role;
DROP TABLE IF EXISTS book_possess;
DROP TABLE IF EXISTS books;
DROP TABLE IF EXISTS book_role_master;
DROP TABLE IF EXISTS booktype_master;

CREATE TABLE booktype_master (
  id                  INTEGER      GENERATED BY DEFAULT AS IDENTITY
 ,booktype_cd         CHAR(03)
 ,booktype            TEXT
 ,selectable          BOOLEAN
 ,user_id             UUID         NOT NULL
                                   REFERENCES auth.users (id)
 ,PRIMARY KEY (booktype_cd, user_id)
);

CREATE TABLE book_role_master (
  id                  INTEGER      GENERATED BY DEFAULT AS IDENTITY
 ,role_cd             CHAR(03)
 ,role_name           TEXT
 ,selectable          BOOLEAN
 ,user_id             UUID         NOT NULL
                                   REFERENCES auth.users (id)
 ,PRIMARY KEY (role_cd, user_id)
);

CREATE TABLE books (
  book_id             INTEGER      GENERATED BY DEFAULT AS IDENTITY
                                   (START WITH 10001)
                                   PRIMARY KEY
 ,isbn10              VARCHAR(10)
 ,isbn13              VARCHAR(13)
 ,c_cd                VARCHAR(05)
 ,ndc                 VARCHAR(10)
 ,title               TEXT         NOT NULL
 ,original_title      TEXT
 ,colophon            TEXT
 ,publisher           TEXT         NOT NULL
 ,publish_series      TEXT
 ,publish_series_no   TEXT
 ,first_publish_year  INTEGER      NOT NULL
 ,remarks             TEXT
 ,comic_f             BOOLEAN
 ,image_url           TEXT
 ,user_id             UUID         NOT NULL
                                   REFERENCES auth.users (id)
 ,UNIQUE (title, publisher, first_publish_year, user_id)
);

CREATE TABLE book_possess (
  book_possess_id     INTEGER      GENERATED BY DEFAULT AS IDENTITY
                                   PRIMARY KEY
 ,book_id             INTEGER      REFERENCES books (book_id)
                                   ON DELETE CASCADE
 ,booktype_cd         CHAR(03)
 ,get_date            DATE         NOT NULL
 ,dispose_date        DATE
 ,image_url           TEXT
 ,remarks             TEXT
 ,user_id             UUID         NOT NULL
                                   REFERENCES auth.users (id)
 ,UNIQUE (book_id, booktype_cd, get_date, user_id)
 ,FOREIGN KEY (booktype_cd, user_id) REFERENCES booktype_master (booktype_cd, user_id)
);

CREATE TABLE book_role (
  id                  INTEGER      GENERATED BY DEFAULT AS IDENTITY
                                   PRIMARY KEY
 ,book_id             INTEGER      REFERENCES books (book_id)
                                   ON DELETE CASCADE
 ,role_cd             CHAR(03)
 ,role_order          INTEGER
 ,person_name         TEXT
 ,remarks             TEXT
 ,user_id             UUID         NOT NULL
                                   REFERENCES auth.users (id)
 ,UNIQUE (book_id, role_cd, person_name, user_id)
 ,FOREIGN KEY (role_cd, user_id) REFERENCES book_role_master (role_cd, user_id)
);

CREATE TABLE book_note (
  id                  INTEGER      GENERATED BY DEFAULT AS IDENTITY
                                   PRIMARY KEY
 ,book_id             INTEGER      REFERENCES books (book_id)
 ,read_st_date        DATE
 ,read_ed_date        DATE
 ,note                TEXT
 ,user_id             UUID         NOT NULL
                                   REFERENCES auth.users (id)
 ,UNIQUE (book_id, read_st_date, user_id)
);

CREATE TABLE profile (
  id                  INTEGER      GENERATED BY DEFAULT AS IDENTITY
 ,user_id             UUID         PRIMARY KEY
 ,user_name           TEXT         NOT NULL
 ,profile_icon        BOOLEAN
);
