-- ■書籍閲覧、一覧
CREATE OR REPLACE FUNCTION search_books_complex(
  p_isbn13 TEXT DEFAULT NULL,
  p_title TEXT DEFAULT NULL,
  p_title_search_type TEXT DEFAULT 'top', -- 'top' or 'part'
  p_publisher TEXT DEFAULT NULL,
  p_publish_series TEXT DEFAULT NULL,
  p_role_cd TEXT DEFAULT NULL,
  p_person_name TEXT DEFAULT NULL,
  p_person_search_type TEXT DEFAULT 'top', -- 'top' or 'part'
  p_bookclass_cd TEXT DEFAULT NULL,
  p_bookform_cd TEXT DEFAULT NULL,
  p_limit_possess TEXT DEFAULT 'noLimit',   -- 'possess', 'nonPossess', 'noLimit'
  p_display_order TEXT DEFAULT 'publish',   -- 'publish', 'get'
  p_select_limit INT DEFAULT 0  -- 0:no limit, 1~:LIMIT
)
RETURNS TABLE (
  book_id INTEGER,
  isbn13 VARCHAR(13),
  title TEXT,
  publisher TEXT,
  publish_series TEXT,
  first_publish_year INTEGER,
  bookclass_cd CHAR(03),
  roles JSONB,
  possessions JSONB
) 
LANGUAGE plpgsql
AS $$
BEGIN
  RETURN QUERY
  SELECT 
    b.book_id, b.isbn13, b.title, b.publisher, b.publish_series, b.first_publish_year, b.bookclass_cd,
    COALESCE(jsonb_agg(DISTINCT jsonb_build_object('role_cd', br.role_cd, 'person_name', br.person_name)) FILTER (WHERE br.id IS NOT NULL), '[]') as roles,
    COALESCE(jsonb_agg(DISTINCT jsonb_build_object('bookform_cd', bp.bookform_cd, 'get_date', bp.get_date, 'dispose_date', bp.dispose_date)) FILTER (WHERE bp.book_possess_id IS NOT NULL), '[]') as possessions
  FROM books b
  LEFT JOIN book_role br ON b.book_id = br.book_id
  LEFT JOIN book_possess bp ON b.book_id = bp.book_id
  WHERE 
    -- 基本フィルタ
    (p_isbn13 IS NULL OR b.isbn13 = p_isbn13)
    AND (p_bookclass_cd IS NULL OR b.bookclass_cd = p_bookclass_cd)
    -- 出版社、出版シリーズ：前方一致
    AND (p_publisher IS NULL OR b.publisher ILIKE p_publisher || '%')
    AND (p_publish_series IS NULL OR b.publish_series ILIKE p_publish_series || '%')
    -- タイトル：前方一致 or 部分一致
    AND (p_title IS NULL OR (CASE WHEN p_title_search_type = 'top' THEN b.title ILIKE p_title || '%' ELSE b.title ILIKE '%' || p_title || '%' END))
    -- 役割 ：存在する場合のみ絞り込む = !inner相当
    AND (p_role_cd IS NULL OR br.role_cd = p_role_cd)
    AND (p_person_name IS NULL OR (CASE WHEN p_person_search_type = 'top' THEN br.person_name ILIKE p_person_name || '%' ELSE br.person_name ILIKE '%' || p_person_name || '%' END))
    -- 保有状況
    AND (p_bookform_cd IS NULL OR bp.bookform_cd = p_bookform_cd)
    AND (CASE 
          WHEN p_limit_possess = 'possess' THEN 
            -- 「未処分」が1つ以上存在する
            EXISTS (SELECT 1 FROM book_possess bp2 WHERE bp2.book_id = b.book_id AND bp2.dispose_date IS NULL)
          WHEN p_limit_possess = 'nonPossess' THEN 
            -- 「未処分」が1つも存在しない (未所持 or 全処分)
            NOT EXISTS (SELECT 1 FROM book_possess bp2 WHERE bp2.book_id = b.book_id AND bp2.dispose_date IS NULL)
          ELSE true END)
  GROUP BY b.book_id
  ORDER BY 
    CASE WHEN p_display_order = 'publish' THEN b.first_publish_year ELSE NULL END ASC,
    CASE WHEN p_display_order <> 'publish' THEN MIN(bp.get_date) ELSE NULL END ASC,
    b.publisher ASC, 
    b.isbn13 ASC,
    b.isbn10 ASC,
    b.title ASC
  LIMIT (CASE WHEN p_select_limit <= 0 THEN NULL ELSE p_select_limit END);
END;
$$;


-- ■未読一覧
CREATE OR REPLACE FUNCTION search_books_unread(
  p_isbn13 TEXT DEFAULT NULL,
  p_title TEXT DEFAULT NULL,
  p_title_search_type TEXT DEFAULT 'top', -- 'top' or 'part'
  p_publisher TEXT DEFAULT NULL,
  p_publish_series TEXT DEFAULT NULL,
  p_role_cd TEXT DEFAULT NULL,
  p_person_name TEXT DEFAULT NULL,
  p_person_search_type TEXT DEFAULT 'top', -- 'top' or 'part'
  p_bookclass_cd TEXT DEFAULT NULL,
  p_bookform_cd TEXT DEFAULT NULL,
  p_display_order TEXT DEFAULT 'publish',   -- 'publish', 'get'
  p_select_limit INT DEFAULT 0  -- 0:no limit, 1~:LIMIT
)
RETURNS TABLE (
  book_id INTEGER
) 
LANGUAGE plpgsql
AS $$
BEGIN
  RETURN QUERY
  SELECT 
    b.book_id
  FROM books b
  -- 1. 保有していることを確認 (RLSにより自分の所有分のみ対象)
  JOIN book_possess bp ON b.book_id = bp.book_id
  WHERE 
    -- 2. ノートが「存在しない」ことを確認 (NOT EXISTSを使用)
    NOT EXISTS (
      SELECT 1 FROM book_note bn WHERE bn.book_id = b.book_id
    )
    -- 3. 役割/著者検索が必要な場合のみ結合
    AND (
      (p_role_cd IS NULL AND p_person_name IS NULL) OR
      EXISTS (
        SELECT 1 FROM book_role br 
        WHERE br.book_id = b.book_id
          AND (p_role_cd IS NULL OR br.role_cd = p_role_cd)
          AND (p_person_name IS NULL OR (CASE WHEN p_person_search_type = 'top' THEN br.person_name ILIKE p_person_name || '%' ELSE br.person_name ILIKE '%' || p_person_name || '%' END))
      )
    )
    -- 4. その他のフィルタ
    AND (p_isbn13 IS NULL OR b.isbn13 = p_isbn13)
    AND (p_bookclass_cd IS NULL OR b.bookclass_cd = p_bookclass_cd)
    -- 出版社、出版シリーズ：前方一致
    AND (p_publisher IS NULL OR b.publisher ILIKE p_publisher || '%')
    AND (p_publish_series IS NULL OR b.publish_series ILIKE p_publish_series || '%')
    -- タイトル：前方一致 or 部分一致
    AND (p_title IS NULL OR (CASE WHEN p_title_search_type = 'top' THEN b.title ILIKE p_title || '%' ELSE b.title ILIKE '%' || p_title || '%' END))
    -- 保有状況
    AND (p_bookform_cd IS NULL OR bp.bookform_cd = p_bookform_cd)
  GROUP BY b.book_id
  ORDER BY 
    CASE WHEN p_display_order = 'publish' THEN b.first_publish_year ELSE NULL END ASC,
    CASE WHEN p_display_order <> 'publish' THEN MIN(bp.get_date) ELSE NULL END ASC,
    b.publisher ASC, 
    b.isbn13 ASC,
    b.isbn10 ASC,
    b.title ASC
  LIMIT (CASE WHEN p_select_limit <= 0 THEN NULL ELSE p_select_limit END);
END;
$$;


-- ■ノート一覧（範囲指定）
CREATE OR REPLACE FUNCTION search_note_range(
  p_read_st_from DATE DEFAULT '0001-01-01',
  p_read_st_to DATE DEFAULT '9999-12-31',
  p_isbn13 TEXT DEFAULT NULL,
  p_title TEXT DEFAULT NULL,
  p_title_search_type TEXT DEFAULT 'top', -- 'top' or 'part'
  p_publisher TEXT DEFAULT NULL,
  p_publish_series TEXT DEFAULT NULL,
  p_role_cd TEXT DEFAULT NULL,
  p_person_name TEXT DEFAULT NULL,
  p_person_search_type TEXT DEFAULT 'top', -- 'top' or 'part'
  p_bookclass_cd TEXT DEFAULT NULL,
  p_bookform_cd TEXT DEFAULT NULL,
  p_select_limit INT DEFAULT 0  -- 0:no limit, 1~:LIMIT
)
RETURNS TABLE (
  note_id INTEGER,
  book_id INTEGER,
  read_st_date DATE,
  read_ed_date DATE,
  note TEXT,
  isbn13 VARCHAR(13),
  title TEXT,
  publisher TEXT,
  publish_series TEXT,
  first_publish_year INTEGER,
  bookclass_cd CHAR(03),
  role_cd CHAR(03),
  person_name TEXT
) 
LANGUAGE plpgsql
AS $$
BEGIN
  RETURN QUERY
  SELECT DISTINCT
    bn.id AS note_id,
    bn.book_id,
    bn.read_st_date,
    bn.read_ed_date,
    bn.note,
    b.isbn13,
    b.title,
    b.publisher,
    b.publish_series,
    b.first_publish_year,
    b.bookclass_cd,
    br_first.role_cd,
    br_first.person_name
  FROM book_note bn
  JOIN books b ON bn.book_id = b.book_id
  -- 関連テーブルは条件がある場合のみ絞り込めるようLEFT JOIN
  LEFT JOIN book_role br ON b.book_id = br.book_id
  LEFT JOIN book_possess bp ON b.book_id = bp.book_id
  -- book_roleの最初の1行
  LEFT JOIN LATERAL (
    SELECT sub_br.role_cd, sub_br.person_name
    FROM book_role sub_br
    WHERE sub_br.book_id = b.book_id
    ORDER BY sub_br.role_cd ASC, sub_br.role_order ASC
    LIMIT 1
  ) br_first ON TRUE
  WHERE
    -- 読書開始日の範囲指定（必須想定）
    bn.read_st_date >= p_read_st_from AND bn.read_st_date < p_read_st_to
    -- 書籍（単純条件）
    AND (p_isbn13 IS NULL OR b.isbn13 = p_isbn13)
    AND (p_bookclass_cd IS NULL OR b.bookclass_cd = p_bookclass_cd)
    -- 出版社、出版シリーズ：前方一致
    AND (p_publisher IS NULL OR b.publisher ILIKE p_publisher || '%')
    AND (p_publish_series IS NULL OR b.publish_series ILIKE p_publish_series || '%')
    -- タイトル：前方一致 or 部分一致
    AND (p_title IS NULL OR (CASE WHEN p_title_search_type = 'top' THEN b.title ILIKE p_title || '%' ELSE b.title ILIKE '%' || p_title || '%' END))
    -- 役割
    AND (p_role_cd IS NULL OR br.role_cd = p_role_cd)
    AND (p_person_name IS NULL OR (CASE WHEN p_person_search_type = 'top' THEN br.person_name ILIKE p_person_name || '%' ELSE br.person_name ILIKE '%' || p_person_name || '%' END))
    -- 保有状況
    AND (p_bookform_cd IS NULL OR bp.bookform_cd = p_bookform_cd)
  ORDER BY bn.read_st_date ASC,
    bn.id ASC
  LIMIT (CASE WHEN p_select_limit <= 0 THEN NULL ELSE p_select_limit END);
END;
$$;


-- ■書籍複製
CREATE OR REPLACE FUNCTION duplicate_book(p_src_book_id INT)
RETURNS INT
LANGUAGE plpgsql
AS $$
DECLARE
  v_new_book_id INT;
  v_timestamp_suffix TEXT;
BEGIN
  v_timestamp_suffix := '_' || to_char(NOW(), 'YYYYMMDDHH24MISS');
  -- (1) booksを複製（書名にタイムスタンプ付加；UNIQUE制約回避）
  INSERT INTO books (
      isbn10, isbn13, c_cd, ndc, title, original_title, colophon, 
      publisher, publish_series, publish_series_no, first_publish_year, 
      bookclass_cd, remarks, image_url, user_id
  )
  SELECT 
      isbn10, isbn13, c_cd, ndc, title || v_timestamp_suffix, original_title, colophon, 
      publisher, publish_series, publish_series_no, first_publish_year, 
      bookclass_cd, remarks, image_url, user_id
  FROM books
  WHERE book_id = p_src_book_id
  RETURNING book_id INTO v_new_book_id; -- コピー先book_id
  IF v_new_book_id IS NULL THEN
      RAISE EXCEPTION '指定された book_id (%) が見つかりません。', p_src_book_id;
  END IF;
  -- (2) book_roleを複製
  INSERT INTO book_role (
      book_id, role_cd, role_order, person_name, remarks, user_id
  )
  SELECT 
      v_new_book_id, role_cd, role_order, person_name, remarks, user_id
  FROM book_role
  WHERE book_id = p_src_book_id;
  -- (3) book_possessを複製
  INSERT INTO book_possess (
      book_id, bookform_cd, get_date, dispose_date, image_url, remarks, user_id
  )
  SELECT 
      v_new_book_id, bookform_cd, get_date, dispose_date, image_url, remarks, user_id
  FROM book_possess
  WHERE book_id = p_src_book_id;
  -- (4) コピー先の book_id を返却
  RETURN v_new_book_id;
END;
$$;


-- ■書籍分類マスタ・分類コード変更用
CREATE OR REPLACE FUNCTION update_bookclass_cd(
  old_cd CHAR(3),
  new_cd CHAR(3),
  new_name TEXT,
  new_selectable BOOLEAN
)
RETURNS void
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
DECLARE
  v_current_user uuid;
  v_avoid_error text;  -- コードを変更して分類名が元のままの時、分類名の重複エラーを避けるため一時的に付加する文字列
BEGIN
  v_current_user := auth.uid();
  v_avoid_error := to_char(NOW(), 'YYYYMMDDHH24MISS');
  IF v_current_user IS NULL THEN
      RAISE EXCEPTION 'ユーザーがログインしていません。';
  END IF;
  -- (1) 書籍分類マスタに新コードを追加（分類名にエラー避けの文字列付加）
  INSERT INTO bookclass_master (bookclass_cd, bookclass, selectable, user_id)
  VALUES (new_cd, new_name || v_avoid_error, new_selectable, v_current_user);
  -- (2) 書籍の分類コードを書換え
  UPDATE books SET bookclass_cd = new_cd
  WHERE  bookclass_cd = old_cd AND user_id = v_current_user;
  -- (3) 書籍分類マスタから旧コードを削除
  DELETE FROM bookclass_master
  WHERE bookclass_cd = old_cd AND user_id = v_current_user;
  -- (4) 新コードの書籍分類マスタの分類名からエラー避けを除去
  UPDATE bookclass_master
  SET bookclass = REGEXP_REPLACE(bookclass, v_avoid_error || '$', '', 'g')
  WHERE bookclass_cd = new_cd AND user_id = v_current_user;
END;
$$;

-- ■データ保存
CREATE OR REPLACE FUNCTION public.backup_books()
RETURNS void
LANGUAGE plpgsql
SECURITY DEFINER -- ログインユーザーの権限に関わらずテーブル操作を実行するため
AS $$
DECLARE
  v_current_user uuid;
  v_history_count integer;
  v_oldest_id integer;
BEGIN
  v_current_user := auth.uid();
  IF v_current_user IS NULL THEN
      RAISE EXCEPTION 'ユーザーがログインしていません。';
  END IF;
  -- (1) スキーマ 'backup2' のテーブルからログインユーザのデータを削除
  CREATE SCHEMA IF NOT EXISTS backup2;
  CREATE TABLE IF NOT EXISTS backup2.bookclass_master (
    id                  INTEGER
   ,bookclass_cd        CHAR(03)
   ,bookclass           TEXT
   ,selectable          BOOLEAN
   ,user_id             UUID
   ,PRIMARY KEY (bookclass_cd, user_id)
  );
  CREATE TABLE IF NOT EXISTS backup2.bookform_master (
    id                  INTEGER
   ,bookform_cd         CHAR(03)
   ,bookform            TEXT
   ,selectable          BOOLEAN
   ,user_id             UUID
   ,PRIMARY KEY (bookform_cd, user_id)
  );
  CREATE TABLE IF NOT EXISTS backup2.book_role_master (
    id                  INTEGER
   ,role_cd             CHAR(03)
   ,role_name           TEXT
   ,selectable          BOOLEAN
   ,user_id             UUID
   ,PRIMARY KEY (role_cd, user_id)
  );
  CREATE TABLE IF NOT EXISTS backup2.books (
    book_id             INTEGER      PRIMARY KEY
   ,isbn10              VARCHAR(10)
   ,isbn13              VARCHAR(13)
   ,c_cd                VARCHAR(05)
   ,ndc                 VARCHAR(10)
   ,title               TEXT
   ,original_title      TEXT
   ,colophon            TEXT
   ,publisher           TEXT
   ,publish_series      TEXT
   ,publish_series_no   TEXT
   ,first_publish_year  INTEGER
   ,bookclass_cd        CHAR(03)
   ,remarks             TEXT
   ,image_url           TEXT
   ,user_id             UUID
   ,UNIQUE (title, publisher, first_publish_year, user_id)
  );
  CREATE TABLE IF NOT EXISTS backup2.book_possess (
    book_possess_id     INTEGER      PRIMARY KEY
   ,book_id             INTEGER
   ,bookform_cd         CHAR(03)
   ,get_date            DATE
   ,dispose_date        DATE
   ,image_url           TEXT
   ,remarks             TEXT
   ,user_id             UUID
   ,UNIQUE (book_id, bookform_cd, get_date, user_id)
  );
  CREATE TABLE IF NOT EXISTS backup2.book_role (
    id                  INTEGER      PRIMARY KEY
   ,book_id             INTEGER
   ,role_cd             CHAR(03)
   ,role_order          INTEGER
   ,person_name         TEXT
   ,remarks             TEXT
   ,user_id             UUID
   ,UNIQUE (book_id, role_cd, person_name, user_id)
  );
  CREATE TABLE IF NOT EXISTS backup2.book_note (
    id                  INTEGER      PRIMARY KEY
   ,book_id             INTEGER
   ,read_st_date        DATE
   ,read_ed_date        DATE
   ,note                TEXT
   ,user_id             UUID
   ,UNIQUE (book_id, read_st_date, user_id)
  );
  CREATE TABLE IF NOT EXISTS backup2.system_constants (
    id                  INTEGER
   ,constant_name       TEXT
   ,constant_type       TEXT
   ,constant_value      TEXT
   ,remarks             TEXT
   ,user_id             UUID
   ,PRIMARY KEY (constant_name, user_id)
  );
  CREATE TABLE IF NOT EXISTS backup2.profile (
    id                  INTEGER
   ,user_id             UUID         PRIMARY KEY
   ,user_name           TEXT
   ,profile_icon        BOOLEAN
  );
  DELETE FROM backup2.profile WHERE user_id = v_current_user;
  DELETE FROM backup2.system_constants WHERE user_id = v_current_user;
  DELETE FROM backup2.book_note WHERE user_id = v_current_user;
  DELETE FROM backup2.book_role WHERE user_id = v_current_user;
  DELETE FROM backup2.book_possess WHERE user_id = v_current_user;
  DELETE FROM backup2.books WHERE user_id = v_current_user;
  DELETE FROM backup2.book_role_master WHERE user_id = v_current_user;
  DELETE FROM backup2.bookform_master WHERE user_id = v_current_user;
  DELETE FROM backup2.bookclass_master WHERE user_id = v_current_user;
  -- (2) スキーマ 'backup1' の各テーブルから 'backup2' にコピー
  IF EXISTS (SELECT 1 FROM information_schema.tables WHERE table_schema = 'backup1' AND table_name = 'bookclass_master') THEN
        INSERT INTO backup2.bookclass_master (id, bookclass_cd, bookclass, selectable, user_id)
        SELECT id, bookclass_cd, bookclass, selectable, user_id
        FROM backup1.bookclass_master WHERE user_id = v_current_user;
    END IF;
    IF EXISTS (SELECT 1 FROM information_schema.tables WHERE table_schema = 'backup1' AND table_name = 'bookform_master') THEN
        INSERT INTO backup2.bookform_master (id, bookform_cd, bookform, selectable, user_id)
        SELECT id, bookform_cd, bookform, selectable, user_id
        FROM backup1.bookform_master WHERE user_id = v_current_user;
    END IF;
    IF EXISTS (SELECT 1 FROM information_schema.tables WHERE table_schema = 'backup1' AND table_name = 'book_role_master') THEN
        INSERT INTO backup2.book_role_master (id, role_cd, role_name, selectable, user_id)
        SELECT id, role_cd, role_name, selectable, user_id
        FROM backup1.book_role_master WHERE user_id = v_current_user;
    END IF;
    IF EXISTS (SELECT 1 FROM information_schema.tables WHERE table_schema = 'backup1' AND table_name = 'books') THEN
        INSERT INTO backup2.books (book_id, isbn10, isbn13, c_cd, ndc, title, original_title, colophon, publisher, publish_series, publish_series_no, first_publish_year, bookclass_cd, remarks, image_url, user_id)
      SELECT book_id, isbn10, isbn13, c_cd, ndc, title, original_title, colophon, publisher, publish_series, publish_series_no, first_publish_year, bookclass_cd, remarks, image_url, user_id
      FROM backup1.books WHERE user_id = v_current_user;
  END IF;
  IF EXISTS (SELECT 1 FROM information_schema.tables WHERE table_schema = 'backup1' AND table_name = 'book_possess') THEN
      INSERT INTO backup2.book_possess (book_possess_id, book_id, bookform_cd, get_date, dispose_date, image_url, remarks, user_id)
      SELECT book_possess_id, book_id, bookform_cd, get_date, dispose_date, image_url, remarks, user_id
      FROM backup1.book_possess WHERE user_id = v_current_user;
  END IF;

  CREATE SCHEMA IF NOT EXISTS backup1;

  INSERT INTO public.backup_history (data_name, backup_date, user_id)
  VALUES ('books', CURRENT_TIMESTAMP, v_current_user);
  -- (6) 最古のバックアップ履歴を削除（3件以上存在する場合）
  SELECT COUNT(*) INTO v_history_count
  FROM public.backup_history
  WHERE data_name = 'books'
    AND user_id = v_current_user;
  IF v_history_count >= 3 THEN
      SELECT id INTO v_oldest_id
      FROM public.backup_history
      WHERE data_name = 'books'
        AND user_id = v_current_user
      ORDER BY backup_date ASC, id ASC
      LIMIT 1;
      IF v_oldest_id IS NOT NULL THEN

      END IF;
